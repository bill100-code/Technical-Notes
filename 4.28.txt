#include <iostream>
#include <cstdlib>
#include <cctype> // 添加ctype.h用于isdigit
using namespace std;

int const MAXSIZE = 1000;

typedef struct Node {
    char data;
    struct Node* next;
} * node;

typedef struct {
    node head;
} strlinkStack;

typedef struct {
    int data[MAXSIZE];
    int top;
} Sqstack;

void initialStack(Sqstack& s) { s.top = 0; }
void push(Sqstack& s, int e) {
    if (s.top == MAXSIZE) { 
        cerr << "栈满错误!" << endl;
        exit(-1);
    }
    s.data[s.top++] = e;
}

int pop(Sqstack& s) {
    if (s.top == 0) {
        cerr << "栈空错误!" << endl;
        exit(-1);
    }
    return s.data[--s.top];
}

void push(strlinkStack& s, char c) {
    node newnode = (node)malloc(sizeof(struct Node));
    newnode->data = c;
    newnode->next = s.head->next;
    s.head->next = newnode;
}

char pop(strlinkStack& s) {
    if (s.head->next == NULL) {
        cerr << "符号栈空错误!" << endl;
        exit(-1);
    }
    node p = s.head->next;
    s.head->next = p->next;
    char c = p->data;
    free(p);
    return c;
}

char top(const strlinkStack& s) {
    if (s.head->next == NULL) return '\0';
    return s.head->next->data;
}

bool isEmpty(const strlinkStack& s) {
    return s.head->next == NULL;
}

// 简化优先级函数 - 仅使用一个优先级系统
int getPriority(char c) {
    switch(c) {
        case '+': case '-': return 1;
        case '*': case '/': return 2;
        case '(': return 0; // 左括号最低优先级
        default: return -1;
    }
}

int calculate(char op, int left, int right) {
    switch (op) {
        case '+': return left + right;
        case '-': return left - right;
        case '*': return left * right;
        case '/': 
            if (right == 0) {
                cerr << "错误：除零!" << endl;
                exit(-1);
            }
            return left / right;
        default: return 0;
    }
}

bool is_num(char c) {
    return isdigit(c); // 使用标准库函数
}

int traIntoNum(char c) { return c - '0'; }

int main() {
    strlinkStack signalStack;
    Sqstack numStack;
    
    initialStack(numStack);
    signalStack.head = (node)malloc(sizeof(struct Node));
    signalStack.head->next = NULL;
    
    char c;
    int currentNum = 0;       // 用于累积多位数
    bool buildingNum = false; // 是否正在构建数字
    
    while ((c = getchar()) != '\n') {
        if (c == ' ') continue; // 跳过空格
        
        if (is_num(c)) {
            currentNum = currentNum * 10 + traIntoNum(c);
            buildingNum = true;
        } else {
            // 处理之前积累的数字
            if (buildingNum) {
                push(numStack, currentNum);
                currentNum = 0;
                buildingNum = false;
            }
            
            if (c == ')') {
                // 修复1：确保括号匹配
                while (!isEmpty(signalStack) && top(signalStack) != '(') {
                    int right = pop(numStack);
                    int left = pop(numStack);
                    char op = pop(signalStack);
                    push(numStack, calculate(op, left, right));
                }
                
                // 关键修复：检查是否找到左括号
                if (!isEmpty(signalStack) && top(signalStack) == '(') {
                    pop(signalStack); // 弹出左括号
                } else {
                    cerr << "错误：括号不匹配!" << endl;
                    return -1;
                }
            } else if (c == '(') {
                push(signalStack, c);
            } else {
                // 修复2：正确处理运算符优先级
                while (!isEmpty(signalStack) && 
                       top(signalStack) != '(' &&  // 排除左括号
                       getPriority(top(signalStack)) >= getPriority(c)) {
                    int right = pop(numStack);
                    int left = pop(numStack);
                    char op = pop(signalStack);
                    push(numStack, calculate(op, left, right));
                }
                push(signalStack, c);
            }
        }
    }
    
    // 处理最后一个数字
    if (buildingNum) {
        push(numStack, currentNum);
    }
    
    // 处理栈中剩余运算符
    while (!isEmpty(signalStack)) {
        if (top(signalStack) == '(') {
            cerr << "错误：未闭合的括号!" << endl;
            return -1;
        }
        int right = pop(numStack);
        int left = pop(numStack);
        char op = pop(signalStack);
        push(numStack, calculate(op, left, right));
    }
    
    // 输出结果
    if (numStack.top == 1) {
        cout << pop(numStack) << endl;
    } else {
        cerr << "错误：无效的表达式!" << endl;
        return -1;
    }
    
    return 0;
}
